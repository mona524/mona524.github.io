js中的this到底是什么?

发表于 2017-11-20   |   分类于 前端

this的指向在函数定义的时候是确定不了的，只有函数执行的时候才能确定this到底指向谁，实际上this的最终指向的是那个调用它的对象

function a(){
    var user = "mona";
    console.log(this.user); //undefined
    console.log(this); //Window
}
a();   #等同于 window.a();

this最终指向的是调用它的对象，这里的函数a实际是被Window对象所点出来的。
例一：

function Func(){
      this.name='mona'; // this，代指obj
  }
  obj = new Func();
  console.log(obj.name); //mona

例二：

function Func(){
      this.name='mona'; // this，代指obj
      this.show = function(){
          console.log(this.name);   //this.指调用这个函数的obj  mona
      }
  }
obj = new Func();
obj.show()

例三：

userInfo = {
     name:'mona',
     age: 18,
     show:function(){
         console.log(this.name) //mona this指的是userinfo这个obj
     }
}       
userInfo.show()

例四： 

function Func(){
    this.name='mona'; // this，代指obj
    this.show = function(){
        console.log(this);  # obj
        (function(){
            console.log(this); # window,自执行函数没有调用者，即是默认的window
        })()
    }
}
obj = new Func();
obj.show()
